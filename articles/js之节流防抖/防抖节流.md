<!-- 防抖 -->
```js
function debunce(fn, delay) {
  var timer = null
  return  function(){
    var _self = this, args = arguments
    clearTimeout(timer)
    timer=setTimeout(function(){
      fn.apply(_self, args)
    },delay)
  }
}
```

<!-- 节流 -->
```js
function throttle(fn, wait){
  var startDate = new Date()
  return function (){
    var nowTime = new Date().getTime(); 
    if(nowTime - startDate > wait){
      fn()
      startDate = nowTime 
    }
  }
}
```
