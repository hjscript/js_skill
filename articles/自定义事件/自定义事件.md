# JavaScript专题之自定义事件

## js简单实现自定义事件

look代码吧。

```js
	function on(ele,type,fn){
		if(!ele[type]){
			ele[type]=[];
		}
		ele[type].push(fn);
	}
	function off(ele,type,fn){
		var eles=ele[type];
		for (var i = 0; i < eles.length; i++) {
			if(eles[i]==fn){
				eles.splice(i,1);
			}
		}
	}
	function run(ele,type,fn){
		var eles=ele[type];
		for (var i = 0; i < eles.length; i++) {
			eles[i]()
		}
	}
```
